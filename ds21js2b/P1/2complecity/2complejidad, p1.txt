<script>   
 let a = 10;
 let b = 15;
 let  mcm = mcmm(a, b);
  console.log (a, b, mcm); 
  
  
 function mcd( a, b) {
        let temporal;
        while (b != 0) {
            temporal = b;
            b = a % b;
            a = temporal;
        }
        return a;
    }
function mcmm( a, b) {
      
        return (a * b) / mcd(a, b);
    }
